  defp replace_heading(line) do
    if String.starts_with?(line, "## ") do
      title = String.replace(line, "## ", "")
      slug =
        title
        |> String.downcase()
        |> String.replace(~r/[^a-z]+/, "-")
        |> String.trim("-")

      "## [#{title}](##{slug})"
    else
      line
    end
  end
